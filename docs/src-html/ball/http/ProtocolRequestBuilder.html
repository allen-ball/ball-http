086/**
087 * <p>
088 * {@link org.apache.hc.core5.http.HttpRequest} builder for
089 * {@link ProtocolClient#protocol()}.  See the {@code type(Annotation,Class)},
090 * {@code method(Annotation,Method)},
091 * {@code parameter(Annotation,Parameter,...)}, and
092 * {@code parameter(Parameter,...)}  methods for the supported protocol
093 * interface, method, and method parameter {@link Annotation}s and types.
094 * </p>
095 * <p>
096 * Protocol API authors should consider designing protocol methods to throw
097 * {@link org.apache.hc.client5.http.HttpResponseException},
098 * {@link org.apache.hc.client5.http.ClientProtocolException}, and
099 * {@link java.io.IOException}.
100 * </p>
101 * <p>
102 * Supported type (interface) annotations:
103 *
104 * {@include #TYPE_ANNOTATIONS}
105 * </p>
106 * <p>
107 * Supported method annotations:
108 *
109 * {@include #METHOD_ANNOTATIONS}
110 * </p>
111 * <p>
112 * Supported method parameter annotations:
113 *
114 * {@include #PARAMETER_ANNOTATIONS}
115 * </p>
116 * <p>
117 * Supported method parameter types:
118 *
119 * {@include #PARAMETER_TYPES}
120 * </p>
121 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
122 */
123@ToString

126    /**
127     * Supported type (interface) annotations.
128     */
129    public static final Set<Class<? extends Annotation>> TYPE_ANNOTATIONS =
130        Arrays.stream(ProtocolRequestBuilder.class.getDeclaredMethods())
131        .filter(t -> t.getName().equals("type"))
132        .filter(t -> t.getParameterCount() > 0)
133        .filter(t -> Annotation.class.isAssignableFrom(t.getParameterTypes()[0]))
134        .filter(t -> ClassUtils.isAssignable(new Class<?>[] { t.getParameterTypes()[0], Class.class },
135                                             t.getParameterTypes()))
136        .map(t -> t.getParameterTypes()[0].asSubclass(Annotation.class))
137        .collect(toSet());

142    public static final Set<Class<? extends Annotation>> METHOD_ANNOTATIONS =
143        Arrays.stream(ProtocolRequestBuilder.class.getDeclaredMethods())
144        .filter(t -> t.getName().equals("method"))
145        .filter(t -> t.getParameterCount() > 0)
146        .filter(t -> Annotation.class.isAssignableFrom(t.getParameterTypes()[0]))
147        .filter(t -> ClassUtils.isAssignable(new Class<?>[] { t.getParameterTypes()[0], Method.class },
148                                             t.getParameterTypes()))

192    /**
193     * Build a {@link org.apache.hc.core5.http.HttpRequest}
194     * ({@link HttpMessage}) from the protocol interface {@link Method}.
195     *
196     * @param   method          The interface {@link Method}.
197     * @param   argv            The caller's arguments.
198     *
199     * @return  The {@link HttpMessage}.
200     *
201     * @throws  Throwable       If the call fails for any reason.
202     */
203    public HttpMessage build(Method method, Object[] argv) throws Throwable {
204        /*
205         * Process annotations and arguments
206         */
207        process(method.getDeclaringClass(), method, argv);
208        /*
209         * Headers
210         */
211        for (Map.Entry<String,String> entry : headers.entrySet()) {
212            if (! request.containsHeader(entry.getKey())) {
213                request.setHeader(entry.getKey(), entry.getValue());
214            }
215        }
216        /*
217         * URI
218         */
219        if (request instanceof HttpUriRequestBase) {
220            ((HttpUriRequestBase) request).setUri(uri.resolveTemplates(templateValues).build());
221        }
222        /*
223         * Body
224         */
225        HttpEntity entity = null;
226
227        if (body instanceof HttpEntity) {
228            entity = (HttpEntity) body;
229        } else if (body instanceof Form) {
230            entity = new UrlEncodedFormEntity((Form) body);
231        } else if (body != null) {
232            entity = new JSONHttpEntity(body);
233        }
234
235        if (entity != null) {
236            ((HttpEntityContainer) request).setEntity(entity);
237        }
238
239        return request;
240    }

242    private void process(Class<?> type, Method method, Object... argv) throws Throwable {
243        for (Annotation annotation : type.getAnnotations()) {
244            try {
245                if (TYPE_ANNOTATIONS.contains(annotation.annotationType())) {
246                    invoke("type",
247                           new Object[] { annotation, type },
248                           annotation.annotationType(), Class.class);
249                }
250            } catch (NoSuchMethodException exception) {
251                throw new IllegalStateException(String.valueOf(annotation.annotationType()), exception);
252            } catch (IllegalAccessException exception) {
253            } catch (InvocationTargetException exception) {
254                throw exception.getTargetException();
255            }
256        }
257
258        for (Annotation annotation : method.getAnnotations()) {
259            try {
260                if (METHOD_ANNOTATIONS.contains(annotation.annotationType())) {
261                    invoke("method",
262                           new Object[] { annotation, method },
263                           annotation.annotationType(), Method.class);
264                }
265            } catch (NoSuchMethodException exception) {
266                throw new IllegalStateException(String.valueOf(annotation.annotationType()), exception);
267            } catch (IllegalAccessException exception) {
268            } catch (InvocationTargetException exception) {
269                throw exception.getTargetException();
270            }
271        }
272
273        Parameter[] parameters = method.getParameters();
274
275        for (int i = 0; i < parameters.length; i += 1) {
276            process(parameters[i], argv[i]);
277        }
278    }
279
280    private void process(Parameter parameter, Object argument) throws Throwable {
281        Annotation[] annotations = parameter.getAnnotations();
282
283        if (annotations.length > 0) {
284            for (int i = 0; i < annotations.length; i += 1) {
285                process(annotations[i], parameter, argument);
286            }
287        } else {
288            try {
289                invoke("parameter",
290                       new Object[] { parameter, argument },
291                       Parameter.class, parameter.getType());
292            } catch (NoSuchMethodException exception) {
293            } catch (IllegalAccessException exception) {
294            } catch (InvocationTargetException exception) {
295                throw exception.getTargetException();
296            }
297        }
298    }
299
300    private void process(Annotation annotation, Parameter parameter, Object argument) throws Throwable {
301        try {
302            if (PARAMETER_ANNOTATIONS.contains(annotation.annotationType())) {
303                invoke("parameter",
304                       new Object[] { annotation, parameter, argument },
305                       annotation.annotationType(), Parameter.class, parameter.getType());
306            }
307        } catch (NoSuchMethodException exception) {
308            throw new IllegalStateException(String.valueOf(annotation.annotationType()), exception);
309        } catch (IllegalAccessException exception) {
310        } catch (InvocationTargetException exception) {
311            throw exception.getTargetException();
312        }
313    }
314
315    private void invoke(String name, Object[] argv, Class<?>... parameters) throws Throwable {
316        MethodUtils.invokeMethod(this, true, name, argv, parameters);
317    }

652        String name = isNotBlank(annotation.value()) ? annotation.value() : parameter.getName();
653
654        if (argument != null) {
655            if (body == null) {
656                body = new Form();
657            }
658
659            ((Form) body).add(name, String.valueOf(argument));
660        }
661    }

797    private abstract class HttpEntityImpl extends AbstractHttpEntity {
798        protected final Object object;
799
800        protected HttpEntityImpl(ContentType type, Object object) {
801            super(type, null, false);
802
803            this.object = requireNonNull(object, "object");
804        }
805
806        @Override
807        public boolean isRepeatable() { return true; }
808
809        @Override
810        public long getContentLength() { return -1; }
811
812        @Override
813        public InputStream getContent() throws IOException, IllegalStateException {
814            ByteArrayDataSource ds = new ByteArrayDataSource(null, null);
815
816            try (OutputStream out = ds.getOutputStream()) {
817                writeTo(out);
818            }
819
820            return ds.getInputStream();
821        }
822
823        @Override
824        public boolean isStreaming() { return false; }
825
826        @Override
827        public void close() { }
828    }
