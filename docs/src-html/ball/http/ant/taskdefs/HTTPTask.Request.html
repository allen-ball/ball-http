076/**
077 * Abstract {@link.uri http://ant.apache.org/ Ant} base {@link Task} for web
078 * API client tasks.
079 *
080 * {@ant.task}
081 *
082 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
083 */
084@NoArgsConstructor(access = PROTECTED)
085public abstract class HTTPTask extends Task implements AnnotatedAntTask, ClasspathDelegateAntTask, ConfigurableAntTask,
086                                                       HttpRequestInterceptor, HttpResponseInterceptor {
087    private static final String DOT = ".";
088
089    private final HttpClientBuilder builder =
090        HttpClientBuilder.create()
091        .addRequestInterceptorLast(this)
092        .addResponseInterceptorLast(this);
093
094    @Getter @Setter @Accessors(chain = true, fluent = true)
095    private ClasspathUtils.Delegate delegate = null;
096    @Getter @Setter
097    private boolean buffer = false;
098
099    @Override
100    public void init() throws BuildException {
101        super.init();
102        ClasspathDelegateAntTask.super.init();
103        ConfigurableAntTask.super.init();
104    }
105
106    @Override
107    public void execute() throws BuildException {
108        super.execute();
109        AnnotatedAntTask.super.execute();
110    }
111
112    /**
113     * Method to allow subclasses to configure the
114     * {@link HttpClientBuilder}.
115     *
116     * @return  The {@link HttpClientBuilder}.
117     */
118    protected HttpClientBuilder builder() { return builder; }
119
120    @Override
121    public void process(HttpRequest request, EntityDetails details, HttpContext context) throws IOException {
122        if (request instanceof HttpEntityContainer) {
123            HttpEntity entity = ((HttpEntityContainer) request).getEntity();
124
125            if (entity != null) {
126                if (! entity.isRepeatable()) {
127                    if (isBuffer()) {
128                        ((HttpEntityContainer) request).setEntity(new BufferedHttpEntity(entity));
129                    }
130                }
131            }
132        }
133
134        log();
135        log(context);
136        log();
137        log(request);
138    }
139
140    @Override
141    public void process(HttpResponse response, EntityDetails details, HttpContext context) throws IOException {
142        if (response instanceof HttpEntityContainer) {
143            HttpEntity entity = ((HttpEntityContainer) response).getEntity();
144
145            if (entity != null) {
146                if (! entity.isRepeatable()) {
147                    if (isBuffer()) {
148                        ((HttpEntityContainer) response).setEntity(new BufferedHttpEntity(entity));
149                    }
150                }
151            }
152        }
153
154        log();
155        log(context);
156        log();
157        log(response);
158    }
159
160    /**
161     * See {@link #log(String)}.
162     *
163     * @param   context         The {@link HttpContext} to log.
164     */
165    protected void log(HttpContext context) {
166        log(new MapTableModel(new BeanMap(context)));
167    }
168
169    /**
170     * See {@link #log(String)}.
171     *
172     * @param   message         The {@link HttpMessage} to log.
173     */
174    protected void log(HttpMessage message) {
175        /*
176         * if (message instanceof HttpRequest) {
177         *     log(String.valueOf(((HttpRequest) message).getRequestLine()));
178         * }
179         *
180         * if (message instanceof HttpResponse) {
181         *     log(String.valueOf(((HttpResponse) message).getStatusLine()));
182         * }
183         */
184        for (Header header : message.getHeaders()) {
185            log(String.valueOf(header));
186        }
187
188        log(getContentType(message), getHttpEntity(message));
189    }
190
191    private String getContentType(HttpMessage message) {
192        return (message.containsHeader(CONTENT_TYPE) ? message.getFirstHeader(CONTENT_TYPE).getValue() : null);
193    }
194
195    private HttpEntity getHttpEntity(HttpMessage message) {
196        HttpEntity entity = null;
197
198        if (entity == null) {
199            if (message instanceof HttpEntityContainer) {
200                entity = ((HttpEntityContainer) message).getEntity();
201            }
202        }
203
204        return entity;
205    }
206
207    /**
208     * See {@link #log(String)}.
209     *
210     * @param   type            The entity {@code Content-Type} (if
211     *                          specified).
212     * @param   entity          The {@link HttpEntity} to log.
213     */
214    protected void log(String type, HttpEntity entity) {
215        if (entity != null) {
216            if (entity.isRepeatable()) {
217                ReaderWriterDataSource ds = new ReaderWriterDataSource(null, type);
218
219                try (OutputStream out = ds.getOutputStream()) {
220                    entity.writeTo(out);
221                    out.flush();
222
223                    String string = ds.toString();
224
225                    if (! isEmpty(string)) {
226                        log();
227                        log(string);
228                    }
229                } catch (IOException exception) {
230                }
231            } else {
232                log(String.valueOf(entity));
233            }
234        }
235    }
236
237    /**
238     * Abstract {@link.uri http://ant.apache.org/ Ant} base
239     * {@link org.apache.tools.ant.Task} for DELETE, GET, POST, and PUT
240     * operations.
241     *
242     * {@ant.task}
243     */
244    @NoArgsConstructor(access = PROTECTED)
245    protected static abstract class Request extends HTTPTask {
246        private PropertiesImpl properties = null;
247        private URIBuilder builder = new URIBuilder();
248        private final List<NameValuePairImpl> headers = new ArrayList<>();
249
250        @Getter @Setter
251        private String content = null;
252
253        public void setURI(String string) throws URISyntaxException {
254            builder = new URIBuilder(string);
255        }
256
257        public void setCharset(String string) {
258            builder.setCharset(Charset.forName(string));
259        }
260
261        public void setFragment(String string) { builder.setFragment(string); }
262        public void setHost(String string) { builder.setHost(string); }
263        public void setPath(String string) { builder.setPath(string); }
264        public void setPort(Integer integer) { builder.setPort(integer); }
265        public void setQuery(String string) { builder.setCustomQuery(string); }
266        public void setScheme(String string) { builder.setScheme(string); }
267        public void setUserInfo(String string) { builder.setUserInfo(string); }
268
269        public void addConfiguredParameter(NameValuePairImpl parameter) {
270            builder.addParameter(parameter.getName(), parameter.getValue());
271        }
272
273        public void addConfiguredHeader(NameValuePairImpl header) {
274            headers.add(header);
275        }
276
277        public void addText(String text) {
278            setContent((isEmpty(getContent()) ? EMPTY : getContent()) + text);
279        }
280
281        @Override
282        public void init() throws BuildException {
283            super.init();
284
285            String method = getClass().getSimpleName().toUpperCase();
286
287            properties = getPrefixedProperties(method + DOT, getProject().getProperties());
288
289            try {
290                if (properties.containsKey("uri")) {
291                    builder = new URIBuilder(properties.getProperty("uri"));
292                }
293
294                properties.configure(builder);
295
296                for (Map.Entry<?,?> entry : getPrefixedProperties("parameter" + DOT, properties).entrySet()) {
297                    builder.addParameter(entry.getKey().toString(), entry.getValue().toString());
298                }
299            } catch (BuildException exception) {
300                throw exception;
301            } catch (Throwable throwable) {
302                throwable.printStackTrace();
303                throw new BuildException(throwable);
304            }
305        }
306
307        private PropertiesImpl getPrefixedProperties(String prefix, Map<?,?> map) {
308            PropertiesImpl properties = new PropertiesImpl();
309
310            for (Map.Entry<?,?> entry : map.entrySet()) {
311                Object key = entry.getKey();
312                String string = (key != null) ? key.toString() : null;
313
314                if ((! isEmpty(string)) && string.startsWith(prefix)) {
315                    properties.put(string.substring(prefix.length()), entry.getValue());
316                }
317            }
318
319            return properties;
320        }
321
322        /**
323         * Method to construct the {@link HTTPTask}-specific
324         * {@link HttpUriRequest}.
325         *
326         * @return      The {@link HttpUriRequest}.
327         */
328        protected abstract HttpUriRequest request();
329
330        /**
331         * Method to configure the {@link HTTPTask} {@link HttpUriRequest}.
332         * See {@link #execute()} and {@link #request()}.
333         *
334         * @param       request         The {@link HttpUriRequest}.
335         *
336         * @throws      Exception       If an exception is encountered.
337         */
338        protected void configure(HttpUriRequest request) throws Exception {
339            ((HttpUriRequestBase) request).setUri(builder.build());
340
341            addHeaders(request, getPrefixedProperties("header" + DOT, properties).entrySet());
342            addHeaders(request, headers);
343
344            if (! isEmpty(getContent())) {
345                setEntity(request, getContent());
346            } else if (! isEmpty(properties.getProperty("content"))) {
347                setEntity(request, properties.getProperty("content"));
348            }
349        }
350
351        private void addHeaders(HttpRequest request, Iterable<? extends Map.Entry<?,?>> iterable) {
352            for (Map.Entry<?,?> entry : iterable) {
353                request.addHeader(entry.getKey().toString(), entry.getValue().toString());
354            }
355        }
356
357        private void setEntity(HttpUriRequest request, String content) throws Exception {
358            if (! isEmpty(content)) {
359                ((HttpEntityContainer) request).setEntity(new StringEntity(content));
360            }
361        }
362
363        @Override
364        public void execute() throws BuildException {
365            super.execute();
366
367            try (CloseableHttpClient client = builder().build()) {
368                HttpUriRequest request = request();
369
370                configure(request);
371
372                HttpResponse response = client.execute(request);
373            } catch (BuildException exception) {
374                throw exception;
375            } catch (Throwable throwable) {
376                throwable.printStackTrace();
377                throw new BuildException(throwable);
378            }
379        }
380    }
