021import ball.lang.reflect.DefaultInterfaceMethodInvocationHandler;
022import java.lang.reflect.Method;
023import lombok.NonNull;
024import lombok.RequiredArgsConstructor;
025import lombok.ToString;
026import org.apache.hc.client5.http.classic.methods.HttpUriRequest;
027import org.apache.hc.core5.http.HttpMessage;
028import org.apache.hc.core5.http.HttpResponse;
029
030/**
031 * Protocol {@link java.lang.reflect.InvocationHandler} for
032 * {@link ProtocolClient}.  The {@link #invoke(Object,Method,Object[])}
033 * translates the interface {@link Method} invocation into an
034 * {@link HttpMessage} through a {@link ProtocolRequestBuilder}.  If the
035 * {@link Method#getReturnType()} is assignable from the generated
036 * {@link HttpMessage}, then it is simply returned; otherwise, the request
037 * is executed through {@link ProtocolClient#client()} with the
038 * {@link ProtocolClient#context()}.  If the {@link Method#getReturnType()}
039 * is {@link HttpResponse} then the response is returned with no further
040 * processing (and the caller is responsible for consuming any entities).
041 * Otherwise, a {@link ProtocolResponseHandler} is provided to the call.
042 *
043 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
044 */
045@RequiredArgsConstructor @ToString
046public class ProtocolInvocationHandler implements DefaultInterfaceMethodInvocationHandler {
047    @NonNull private final ProtocolClient<?> client;
048
049    @Override
050    public Object invoke(Object proxy, Method method, Object[] argv) throws Throwable {
051        Object result = null;
052        Class<?> declarer = method.getDeclaringClass();
053
054        if (method.isDefault()) {
055            result = DefaultInterfaceMethodInvocationHandler.super.invoke(proxy, method, argv);
056        } else if (declarer.equals(Object.class)) {
057            result = method.invoke(proxy, argv);
058        } else {
059            Class<?> returnType = method.getReturnType();
060            HttpMessage request = new ProtocolRequestBuilder(client).build(method, argv);
061
062            if (returnType.isAssignableFrom(request.getClass())) {
063                result = returnType.cast(request);
064            } else if (returnType.isAssignableFrom(HttpResponse.class)) {
065                result = client.client().execute((HttpUriRequest) request, client.context());
066            } else {
067                ProtocolResponseHandler handler = new ProtocolResponseHandler(client, method);
068
069                result = client.client().execute((HttpUriRequest) request, client.context(), handler);
070            }
071        }
072
073        return result;
074    }
075}
