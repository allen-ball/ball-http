021import ball.annotation.ServiceProviderFor;
022import ball.annotation.processing.AnnotatedProcessor;
023import ball.annotation.processing.For;
024import ball.http.annotation.Protocol;
025import java.lang.annotation.Annotation;
026import java.util.Set;
027import javax.annotation.processing.Processor;
028import javax.annotation.processing.RoundEnvironment;
029import javax.lang.model.element.AnnotationMirror;
030import javax.lang.model.element.AnnotationValue;
031import javax.lang.model.element.Element;
032import javax.lang.model.element.ExecutableElement;
033import javax.lang.model.element.TypeElement;
034import javax.ws.rs.DELETE;
035import javax.ws.rs.GET;
036import javax.ws.rs.HEAD;
037import javax.ws.rs.OPTIONS;
038import javax.ws.rs.PATCH;
039import javax.ws.rs.POST;
040import javax.ws.rs.PUT;
041import lombok.NoArgsConstructor;
042import lombok.ToString;
043
044import static java.util.stream.Collectors.toSet;
045import static javax.tools.Diagnostic.Kind.ERROR;
046import static lombok.AccessLevel.PROTECTED;
047
048/**
049 * Abstract base class for {@link javax.ws.rs JSR 311} {@link Annotation}
050 * {@link Processor} for {@link Protocol} interface methods.
051 *
052 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
053 */
054@NoArgsConstructor(access = PROTECTED)
055public abstract class ProtocolJSR311AnnotationProcessor extends AnnotatedProcessor {
056    @Override
057    public void process(RoundEnvironment roundEnv, TypeElement annotation, Element element) {
058        super.process(roundEnv, annotation, element);
059
060        switch (element.getKind()) {
061        case METHOD:
062            TypeElement type = (TypeElement) element.getEnclosingElement();
063
064            switch (type.getKind()) {
065            case INTERFACE:
066                if (type.getAnnotation(Protocol.class) != null) {
067                    check(annotation, type, (ExecutableElement) element);
068                }
069                break;
070
071            default:
072                break;
073            }
074            break;
075        }
076    }
077
078    /**
079     * Method to check an annotated interface method.
080     *
081     * @param   annotation      The annotation being processed.
082     * @param   type            The interface containing the method.
083     * @param   method          The annotated method.
084     */
085    protected abstract void check(TypeElement annotation,
086                                  TypeElement type, ExecutableElement method);
087
088    /**
089     * {@link javax.ws.rs JSR 311} method {@link Annotation}
090     * {@link Processor}.
091     */
092    @ServiceProviderFor({ Processor.class })
093    @For({ DELETE.class, GET.class, HEAD.class, OPTIONS.class, PATCH.class, POST.class, PUT.class })
094    @NoArgsConstructor @ToString
095    public static class Method extends ProtocolJSR311AnnotationProcessor {
096        @Override
097        protected void check(TypeElement annotation, TypeElement type, ExecutableElement method) {
098            Set<Class<? extends Annotation>> set =
099                getSupportedAnnotationTypeList().stream()
100                .filter(t -> method.getAnnotation(t) != null)
101                .collect(toSet());
102
103            switch (set.size()) {
104            case 0:
105                throw new IllegalStateException();
106                /*
107                 * break;
108                 */
109            case 1:
110                AnnotationMirror mirror = getAnnotationMirror(method, annotation);
111                AnnotationValue value = getAnnotationValue(mirror, "value");
112                break;
113
114            default:
115                print(ERROR, method,
116                      "%s may only be annotated with one of %s",
117                      method.getKind(), set.stream().map(t -> "@" + t.getSimpleName()).collect(toSet()));
118                break;
119            }
120        }
121    }
122}
